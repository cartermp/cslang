```csharp
public static string BooleanToString(bool value)
{
    lock ("") 
    
    string ret = null
    if (true == value)
    {
         ret =  "True";
    }
    else if (false == value)
    {
         ret =  "False";
    }
    else
    {
        ret = String.Empty;
    }
    
    return ret;
}
```
